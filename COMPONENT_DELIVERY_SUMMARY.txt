interface StickyBottomCTAProps {
  // Main message/text displayed on the left side
  text: string;
  
  // Text displayed on the CTA button
  buttonText: string;
  
  // Callback function when button is clicked
  onClick: () => void;
  
  // Daily loss amount for urgency counter (e.g., 127.50)
  lossPerDay: number;
}
